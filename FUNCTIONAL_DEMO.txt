 * =====================
 * 
 * 1. USER CLICKS ADDRESS BAR
 *    → Cursor appears, ready for input

 * 
 * 2. USER TYPES URL
 *    → Example: "google.com"
 *    → Text appears in real-time
 * 
 * 3. USER PRESSES ENTER
 *    → HandleKeyEvent() captures VKEY_RETURN
 *    → Auto-formats to "http://google.com"
 *    → Validates URL
 *    → Navigates to the URL
 * 
 * 4. NAVIGATION HAPPENS
 *    → Progress bar shows at top of address bar
 *    → Page loads in web content area
 *    → Address bar updates with final URL
 * 
 * 5. USER CAN NAVIGATE HISTORY
 *    → Back/Forward buttons work
 *    → Refresh/Stop buttons available
 */

 * CODE TRANSFORMATION SUMMARY:
 * ============================
 */

// OLD Implementation (Display Only):
class AddressView_OLD : public views::Label {
    // Can only display URLs
    // No user interaction
    // Static text display
};

// NEW Implementation (Interactive):
class AddressView_NEW : public views::Textfield, 
                       public views::TextfieldController {
    // Editable text input
    // Enter key navigation  
    // Auto URL formatting
    // Progress visualization
    
    bool HandleKeyEvent(views::Textfield* sender, const ui::KeyEvent& key_event) {
        if (key_event.key_code() == ui::VKEY_RETURN) {
            // Navigate to entered URL
            navigateToURL(GetText());
            return true;
        }
        return false;
    }
};
